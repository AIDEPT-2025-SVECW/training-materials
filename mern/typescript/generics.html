3
* Without generics, we have to use `any` when building reusable code.
4
*/
5
6
// A function that takes an argument and returns it.
7
function echo(arg: any): any {
8
return arg;
9
}
10
11
let result = echo("Hello World");
12
13
// ❌ THE FLAW: We lost the type information!
14
// TS thinks 'result' is of type 'any', not 'string'.
15
// If we try to call result.length, we won't get autocomplete or safety.
16
17
/**
18
* The Solution: Generics
19
* Generics allow us to pass a "Type Variable" into our function.
20
* It captures the type the user provides, and locks it in.
21
*/
22
23
// declares the generic type variable.
24
function genericEcho<T>(arg: T): T {
25
return arg;
26
}
27
28
// ✅ SUCCESS: TS captures that T is a string!
29
let safeResult = genericEcho<string>("Hello World");

31
// Note: Type Inference usually allows you to drop the entirely!
32

/**
2
* Generic Constraints
3
* Sometimes you want a Generic, but you don't want it to be *anything*.
4
* You want to constrain it to guarantee it has certain properties.
5
*/
6
7
interface Lengthwise {
8

9
}
10
11
// We use the `extends` keyword.
12
// T can be anything, AS LONG AS it has a .length property!
13
function logLength<T extends Lengthwise>(arg: T): T {
14
console.log(arg.length); // TS knows .length exists safely!
15
return arg;
16
}
17
18
logLength([1, 2, 3]); // ✅ OK: Arrays have length
19
logLength("Hello"); // ✅ OK: Strings have length
20
logLength(100); // ❌ ERROR: Number does not have length
