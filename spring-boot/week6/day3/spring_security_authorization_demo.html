@Configuration
@EnableWebSecurity
@EnableGlobalMethodSecurity(prePostEnabled = true)
public class SecurityConfig {

    @Autowired
    private DaoAuthenticationProvider authProvider; // Already configured

    @Override
    protected void configure(HttpSecurity http) throws Exception {
        http
            .authorizeRequests()
                .antMatchers("/api/public/**").permitAll()
                .antMatchers("/api/admin/**").hasRole("ADMIN")
                .antMatchers("/api/reports/**").hasRole("MANAGER")
                .antMatchers("/api/profile/**").hasRole("USER")
                .antMatchers(HttpMethod.POST, "/api/documents/**")
                    .hasAuthority("WRITE")
                .antMatchers(HttpMethod.DELETE, "/api/documents/**")
                    .hasAuthority("DELETE")
                .anyRequest().authenticated()
            .and()
                .authenticationProvider(authProvider) // Use existing provider
            .and()
                .sessionManagement()
                .sessionCreationPolicy(SessionCreationPolicy.STATELESS);
    }
}

@PreAuthorize

Check authorization before method execution

@PostAuthorize

Check authorization after method execution

@Secured

Simple role-based method security

@RolesAllowed

JSR-250 standard annotation

@RestController
public class DocumentController {

    @PreAuthorize("hasRole('USER')")
    @GetMapping("/api/documents")
    public List<Document> getDocuments() {
        return documentService.findAll();
    }

    @PreAuthorize("hasAuthority('WRITE')")
    @PostMapping("/api/documents")
    public Document createDocument(@RequestBody Document doc) {
        return documentService.save(doc);
    }

    @PreAuthorize("hasRole('ADMIN') or @documentService.isOwner(#id, authentication.name)")
    @DeleteMapping("/api/documents/{id}")
    public void deleteDocument(@PathVariable Long id) {
        documentService.delete(id);
    }

    @PostAuthorize("returnObject.owner == authentication.name or hasRole('ADMIN')")
    @GetMapping("/api/documents/{id}")
    public Document getDocument(@PathVariable Long id) {
        return documentService.findById(id);
    }

    @Secured({"ROLE_ADMIN", "ROLE_MANAGER"})
    @GetMapping("/api/reports/sensitive")
    public Report getSensitiveReport() {
        return reportService.getSensitiveData();
    }
}
Method-Level Security Benefits:
‚Ä¢ Fine-grained control over individual methods
‚Ä¢ Support for complex authorization logic using SpEL
‚Ä¢ Can combine with URL-based security for defense in depth
‚Ä¢ Access to method parameters and return values in expressions 
// Common SpEL expressions for authorization

// Role-based checks
hasRole('ADMIN')                    // User has ADMIN role
hasAnyRole('ADMIN', 'MANAGER')      // User has any of these roles

// Authority-based checks  
hasAuthority('WRITE')               // User has WRITE authority
hasAnyAuthority('READ', 'WRITE')    // User has any of these authorities

// Authentication object access
authentication.name                 // Current username
authentication.principal            // UserDetails object
authentication.authorities          // Collection of authorities

// Complex expressions
hasRole('ADMIN') or @userService.isOwner(#userId, authentication.name)
hasAuthority('WRITE') and #document.status == 'DRAFT'
authentication.name == #targetUsername or hasRole('ADMIN')

// Method parameter access
@PreAuthorize("#document.owner == authentication.name")
public void updateDocument(Document document) { ... }

// Return value filtering
@PostAuthorize("returnObject.owner == authentication.name")
public Document getDocument(Long id) { ... }

// Collection filtering
@PostFilter("filterObject.owner == authentication.name or hasRole('ADMIN')")
public List<Document> getDocuments() { ... }
